var removeDuplicates = function (s) {
  let stack = [];

  for (let char of s) {
    let lastS = stack.pop();
    if (char !== lastS) {
      stack.push(lastS, char);
    }
  }
  return stack.join("");
};

console.log(removeDuplicates("abbaca")); // 'ca'
console.log(removeDuplicates("azxxzy")); // "ay"
