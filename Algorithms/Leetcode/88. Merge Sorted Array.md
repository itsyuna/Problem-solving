const merge = function (nums1, m, nums2, n) {
  let j = 0;
  for (let i = m; i < m + n; i++) {
    nums1[i] = nums2[j];
    j++;
  }
  nums1.sort((a, b) => {
    if (a > b) return 1;
    if (a < b) return -1;
    return 0;
  });
};

console.log(merge([1, 2, 3, 0, 0, 0], 3, [2, 5, 6], 3)); // [1,2,2,3,5,6]
console.log(merge([1], 1, [], 0)); // [1]
console.log(merge([0], 0, [1], 1)); // [1]
